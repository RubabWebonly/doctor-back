def fix_text(t: str) -> str:
 
    t = t.lower()

    t = re.sub(r"[!?;:]+", ".", t)

  
    t = re.sub(r",(?=\S)", ", ", t)


    t = re.sub(r"\s+", " ", t)


    corrections = {
        "maked": "made",
        "writed": "wrote",
        "dissapear": "disappear",
        "dissapeard": "disappeared"
    }
    for wrong, right in corrections.items():
        t = t.replace(wrong, right)

  
    sentences = [s.strip().capitalize() for s in t.split(".") if s.strip()]


    fixed = ". ".join(sentences) + "."
    return fixed

print(fix_text(text))
