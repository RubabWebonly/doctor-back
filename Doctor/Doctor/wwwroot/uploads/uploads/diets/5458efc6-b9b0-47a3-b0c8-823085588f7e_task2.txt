

df= pd.read_excel("day2_dataset_map_filter_reduce .xlsx")


//1) Map: 'Name' sütununu hamısını böyük hərflərə çevirərək yeni siyahı yaradın.
def map_names_upper(df):
    return df["Name"].str.upper().tolist()

//2) Map: 'SalaryAZN' dəyərlərini 10% artımla (round) yeni siyahıda göstərin.
def map_salary_increase(df):
    return df["SalaryAZN"].apply(lambda x: round(x * 1.1)).tolist()

//3) Filter: Yalnız 'City' = 'Baku' olan əməkdaşları seçin (sətir siyahısı).
def filter_city_baku(df):
    return df[df["City"] == "Baku"]

//4) Filter: 'Age' >= 30 və 'Department' = 'Engineering' olanları çıxarın.
def filter_age_department(df):
    return df[(df["Age"] >= 30) & (df["Department"] == "Engineering")]

//5) Filter: 'Remote' = True və 'Performance' >= 80 olan sətirləri seçin.
def filter_remote_perf(df):
    return df[(df["Remote"] == True) & (df["Performance"] >= 80)]

//6) Reduce: Bütün 'SalaryAZN' toplamını hesablayın.
def reduce_salary_sum(df):
    return df["SalaryAZN"].sum()


//7) Reduce: 'SalaryAZN' üzrə maksimumu və minimumu tapın.
def reduce_salary_minmax(df):
    return df["SalaryAZN"].min(), df["SalaryAZN"].max()


//8) Map+Filter: 'Skills' içində 'python' olan əməkdaşların adlarını böyük hərflərlə qaytarın.
def mapfilter_python_names(df):
    return df[df["Skills"].str.contains("python", case=False)]["Name"].str.upper().tolist()


//9) Map: 'JoinDate' dəyərlərini 'YYYY-MM' formatına salın.
def map_joindate_format(df):
    return pd.to_datetime(df["JoinDate"]).dt.strftime("%Y-%m").tolist()

//10) Filter+Reduce: 'City' = 'Baku' olanların 'SalaryAZN' cəmini hesablayın.
def filterreduce_baku_salary(df):
    return df[df["City"] == "Baku"]["SalaryAZN"].sum()


//11) Map: 'Performance' balına görə kateqoriya təyin edin: 1-59='Low', 60-79='Medium', 80-100='High'.
def map_performance_category(df):
    def category(p):
        if p <= 59: return "Low"
        elif p <= 79: return "Medium"
        else: return "High"
    return df["Performance"].apply(category).tolist()


//12) Filter: 'Skills' içində 'django' VƏ 'docker' olan sətirləri çıxarın.
def filter_django_docker(df):
    return df[df["Skills"].str.contains("django", case=False) & df["Skills"].str.contains("docker", case=False)]


//13) Map: 'Age' → gələn il üçün yaş (Age+1) siyahısını yaradın.
def map_age_plus1(df):
    return (df["Age"] + 1).tolist()

//14) Filter+Map: 'Department' = 'Data' olanların maaşını 15% artırıb yeni siyahı yaradın.
def filtermap_data_salary(df):
    return df[df["Department"] == "Data"]["SalaryAZN"].apply(lambda x: round(x * 1.15)).tolist()


//15) Bonus: 'Engineering' şöbəsində 'python' bilənlərin orta maaşını hesablayın.
def bonus_eng_python_avg(df):
    filtered = df[(df["Department"] == "Engineering") & (df["Skills"].str.contains("python", case=False))]
    return filtered["SalaryAZN"].mean()
